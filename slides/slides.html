Racket – Syntax

3, "String", sym, true                   ;; Atomare Werte

(* (+ 4 3) 2)                            ;; Integer, 14

(list 1 2 3 "Hallo" (+ 1 2))             ;; Liste, (1 2 3 "Hallo" 3)

(if (> 2 3) "It is true" 100)            ;; Integer, 100

(cond
  [(> 3 4) "Ich bin nicht das Ergebnis"]
  [(= 3 3) "Ich bin das Ergebnis"]
  [(> 3 2) "Ich bin nicht das Ergebnis"])  ;; String
