(defn ping-with-retries-m
  [tries timeout address]
  (if (<= tries 0)
    (monad/return false)
    (monad/monadic [response (ping-m timeout address)]
                   (if response
                     (monad/return response)
                     (ping-with-retries-m (dec tries) timeout address)))))
